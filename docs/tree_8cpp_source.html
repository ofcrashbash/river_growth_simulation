26  {

29  auto dl = TipVector().norm();

31  RemoveTipPoint();

51  void TreeVector(vector<Point> &tree_vector, int id, const Tree& tr, double eps)
52  {
53  const BranchNew& br = tr.GetBranch(id);
54  vector<Point> left_vector, right_vector;
55  left_vector.reserve(br.Size());
56  right_vector.reserve(br.Size());
57 
58  for(unsigned i = 0; i < br.Size(); ++i)
59  {
60  if(i == 0)
61  {
62  left_vector.push_back(
63  br.GetPoint(i) + Point{Polar{eps/2, br.SourceAngle() + M_PI/2}});
64  right_vector.push_back(
65  br.GetPoint(i) + Point{Polar{eps/2, br.SourceAngle() - M_PI/2}});
66  }
67  else
68  {
69  left_vector.push_back(
70  br.GetPoint(i) + br.Vector(i).normalize().rotate(M_PI/2)*eps/2);
71  right_vector.push_back(
72  br.GetPoint(i) + br.Vector(i).normalize().rotate(-M_PI/2)*eps/2);
73  }
74  }
75 
76 
77  if(tr.HasSubBranches(id))
78  {
79  auto[left_b_id, right_b_id] = tr.GetSubBranches(id);
80  tree_vector.insert(end(tree_vector),
81  left_vector.begin(), left_vector.end() - 1);
82  TreeVector(tree_vector, left_b_id, tr, eps);
83  tree_vector.pop_back();
84  TreeVector(tree_vector, right_b_id, tr, eps);
85  tree_vector.insert(end(tree_vector),
86  right_vector.rbegin() + 1, right_vector.rend());
87  }
88  else
89  {
90  tree_vector.insert(end(tree_vector),
91  left_vector.begin(), left_vector.end() - 1);
92  tree_vector.push_back(br.GetPoint(br.Size() - 1));
93  tree_vector.insert(end(tree_vector),
94  right_vector.rbegin() + 1, right_vector.rend());
95  }
96  }
