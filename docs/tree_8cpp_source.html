20  BranchNew& BranchNew::Shrink(double lenght)
21  {
22  while(lenght > 0)
23  {
24  auto dl = TipVector().norm();
25  if(lenght >= dl)
26  RemoveTipPoint();
27  else
28  {
29  auto k = lenght/dl;
30  auto new_tip = TipVector()*k;
31  RemoveTipPoint();
32  AddPoint(new_tip);
33  lenght = 0;
34  }
35  lenght -= dl;
36  }

40  }

43  void TreeVector(vector<Point> &tree_vector, int id, Tree& tr, double eps)
44  {
45  BranchNew& br = tr.GetBranch(id);
46  vector<Point> left_vector, right_vector;
47  left_vector.reserve(br.Size());
48  right_vector.reserve(br.Size());
49 
50  for(unsigned i = 0; i < br.Size(); ++i)
51  {
52  if(i == 0)
53  {
54  left_vector.push_back(
55  br.GetPoint(i) + Point{Polar{eps/2, br.SourceAngle() + M_PI/2}});
56  right_vector.push_back(
57  br.GetPoint(i) + Point{Polar{eps/2, br.SourceAngle() - M_PI/2}});
58  }
59  else
60  {
61  left_vector.push_back(
62  br.GetPoint(i) + br.Vector(i).normalize().rotate(M_PI/2)*eps/2);
63  right_vector.push_back(
64  br.GetPoint(i) + br.Vector(i).normalize().rotate(-M_PI/2)*eps/2);
65  }
66  }
67 
68 
69  if(tr.HasSubBranches(id))
70  {
71  auto[left_b_id, right_b_id] = tr.GetSubBranches(id);
72  tree_vector.insert(end(tree_vector),
73  left_vector.begin(), left_vector.end() - 1);
74  TreeVector(tree_vector, left_b_id, tr, eps);
75  tree_vector.pop_back();
76  TreeVector(tree_vector, right_b_id, tr, eps);
77  tree_vector.insert(end(tree_vector),
78  right_vector.rbegin() + 1, right_vector.rend());
79  }
80  else
81  {
82  tree_vector.insert(end(tree_vector),
83  left_vector.begin(), left_vector.end() - 1);
84  tree_vector.push_back(br.GetPoint(br.Size() - 1));
85  tree_vector.insert(end(tree_vector),
86  right_vector.rbegin() + 1, right_vector.rend());
87  }
88  }
