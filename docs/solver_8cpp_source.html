160 void Solver::assemble_system()
161 {
162  const QGauss<dim> quadrature_formula(3);
163  FEValues<dim> fe_values(fe, quadrature_formula,
164  update_values | update_gradients |
165  update_quadrature_points | update_JxW_values);
166  const unsigned dofs_per_cell = fe.dofs_per_cell;
167  const unsigned n_q_points = quadrature_formula.size();
168  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
169  Vector<double> cell_rhs(dofs_per_cell);
170  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
171  typename DoFHandler<dim>::active_cell_iterator
172  cell = dof_handler.begin_active(),
173  endc = dof_handler.end();
174  for (; cell != endc; ++cell)
175  {
176  cell_matrix = 0;
177  cell_rhs = 0;
178  fe_values.reinit(cell);
179  for (unsigned q_index = 0; q_index < n_q_points; ++q_index)
180  {
181  const double current_coefficient = coefficient(fe_values.quadrature_point(q_index));
182  for (unsigned i = 0; i < dofs_per_cell; ++i)
183  {
184  for (unsigned int j = 0; j < dofs_per_cell; ++j)
185  cell_matrix(i, j) += (current_coefficient *
186  fe_values.shape_grad(i, q_index) *
187  fe_values.shape_grad(j, q_index) *
188  fe_values.JxW(q_index));
189  cell_rhs(i) += (fe_values.shape_value(i, q_index) *
190  1.0 *
191  fe_values.JxW(q_index));
192  }
193  }
194  cell->get_dof_indices(local_dof_indices);
195  constraints.distribute_local_to_global(cell_matrix,
196  cell_rhs,
197  local_dof_indices,
198  system_matrix,
199  system_rhs);
200  }
