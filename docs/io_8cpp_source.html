62  {

66  ofstream out(file_name);

68 
69 
70  //Branches
71  json branches;
72  for(auto id: tr.branches_index)
73  {
74  auto branch_id = id.first;
75  auto& branch = tr.GetBranch(branch_id);
76  vector<pair<double, double>> coords(branch.Size());
77  for(unsigned i = 0; i < branch.Size(); ++i)
78  coords[i] = {branch.GetPoint(i).x, branch.GetPoint(i).y};
79 
80 
81  branches.push_back({
82  {"source_point", {branch.SourcePoint().x , branch.SourcePoint().y}},
83  {"source_angle", branch.SourceAngle()},
84  {"coords", coords},
85  {"id", branch_id}});
86  }
87 
88  //Border
89  json jborder;
90  {
91  vector<pair<double, double>> coords;
92  vector<vector<int>> lines;
93  coords.reserve(border.vertices.size());
94  coords.reserve(border.lines.size());
95  for(auto& p: border.vertices)
96  coords.push_back({p.x, p.y});
97  for(auto& l: border.lines)
98  lines.push_back({(int)l.p1, (int)l.p2, l.id});
99  jborder = {
100  {"SourceIndexes", border.sources},
101  {"SomeDetails", "points and lines should be in counterclockwise order!"},
102  {"coords", coords},
103  {"lines", lines}};
104  }
105 
106 
107  //implementation with json
108  json j = {
109  {"Description", "This is RiverSim simulation data"},
110 
111  {"RuntimeInfo", {
112  {"StartDate", time.CreationtDate()},

114  {"TotalTime", time.Total()},
115  {"EachCycleTime", time.records},
116  {"InputFile", "TODO"}}},
117 
118  {"Model", {

120  {"width", mdl.width},
121  {"height", mdl.height},
122  {"river-boundary-id", mdl.river_boundary_id},

134 
135  {"Integration",{

138 
139  {"Mesh", {

159  void Open(Model& mdl, Border& border, Tree& tr, string file_name)
160  {
161  ifstream in(file_name);
162  if(!in) throw invalid_argument("Open. Can't create file for read.");
163 
164  json j;
165  in >> j;
166  if(j.count("Model"))
167  {

174  jmdl.at("boundary-ids").get_to(mdl.boundary_ids);
175 
176  jmdl.at("boundary-condition").get_to(mdl.boundary_condition);
177  jmdl.at("field-value").get_to(mdl.field_value);

180  jmdl.at("biffurcation-threshold").get_to(mdl.biffurcation_threshold);
181  jmdl.at("biffurcation-angle").get_to(mdl.biffurcation_angle);
182  jmdl.at("biffurcation-threshold").get_to(mdl.biffurcation_threshold);
183  jmdl.at("growth-type").get_to(mdl.growth_type);
184  jmdl.at("growth-threshold").get_to(mdl.growth_threshold);
185  jmdl.at("ds").get_to(mdl.ds);
186 
187  if(jmdl.count("Mesh"))
188  {
189  auto jmesh = jmdl["Mesh"];
190 
191  jmesh.at("eps").get_to(mdl.mesh.eps);
192  jmesh.at("exponant").get_to(mdl.mesh.exponant);
193  jmesh.at("max-area").get_to(mdl.mesh.max_area);
194  jmesh.at("min-area").get_to(mdl.mesh.min_area);
195  jmesh.at("min-angle").get_to(mdl.mesh.min_angle);
196  jmesh.at("refinment-radius").get_to(mdl.mesh.refinment_radius);
197  }
198  if(jmdl.count("Integration"))
199  {
200  auto jinteg = jmdl["Integration"];
201 
202  jinteg.at("radius").get_to(mdl.integr.integration_radius);
203  jinteg.at("exponant").get_to(mdl.integr.exponant);
204  }
205  }
206  if(j.count("Trees"))
207  {
208  auto jtrees = j["Trees"];
209  jtrees.at("SourceIds").get_to(tr.source_branches_id);
210  jtrees.at("Relations").get_to(tr.branches_relation);
211 
212 
213  for (auto& [key, value] : jtrees["Branches"].items())
214  {
215  vector<double> s_point;
216  vector<pair<double, double>> coords;
217  double source_angle;
218  int id;
219  value.at("source_point").get_to(s_point);
220  value.at("source_angle").get_to(source_angle);
221  value.at("coords").get_to(coords);
222  value.at("id").get_to(id);
223 
224  BranchNew branch(River::Point{s_point.at(0), s_point.at(1)}, source_angle);
225  branch.points.resize(coords.size());
226  for(unsigned i = 0; i < coords.size(); ++i)
227  {
228  branch.points[i] = River::Point{coords.at(i).first, coords.at(i).second};
229  }
230  tr.AddBranch(branch, id);
231  }
232 
233  }
234  if(j.count("Border"))
235  {
236  auto jborder = j["Border"];
237  vector<pair<double, double>> coords;
238  vector<vector<int>> lines;
239 
240  jborder.at("SourceIndexes").get_to(border.sources);
241  jborder.at("coords").get_to(coords);
242  jborder.at("lines").get_to(lines);
243 
244  border.vertices.reserve(coords.size());
245  border.lines.reserve(lines.size());
246  for(auto &p: coords)
247  border.vertices.push_back({p.first, p.second});
248 
249  for(auto &l: lines)
250  border.lines.push_back({(long unsigned)l.at(0)/*p1*/, (long unsigned)l.at(1)/*p2*/, l.at(2)/*id*/});
251  }
252  }
