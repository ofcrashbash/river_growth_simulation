53  {

55 
56  //FIXME:

64  //Deal.II library
65  sim.SetBoundaryRegionValue(mdl.GetZeroIndices(), 0.);
66  sim.SetBoundaryRegionValue(mdl.GetNonZeroIndices(), 1.);
67  sim.OpenMesh(mesh_file);
68  sim.run(0/*FIXME*/);
69 
70  for(auto id: tree.TipBranchesId())
71  {
72  auto tip_point = tree.GetBranch(id).TipPoint();
73  auto tip_angle = tree.GetBranch(id).TipAngle();
74  auto series_params = sim.integrate(mdl, tip_point, tip_angle);
75 
76  if(mdl.q_growth(series_params))
77  {
78  if(mdl.q_biffurcate(series_params))
79  {
80  auto br_left = BranchNew(tip_point, tip_angle + mdl.biffurcation_angle);
81  br_left.AddPoint(Polar{mdl.ds, 0});
82  auto br_right = BranchNew(tip_point, tip_angle - mdl.biffurcation_angle);
83  br_right.AddPoint(Polar{mdl.ds, 0});
84  tree.AddSubBranches(id, br_left, br_right);
85  }
86  else

88  }
