21  Border& Border::MakeRectangular(
22  vector<double> regionSize, vector<int> boundariesId,
23  vector<double> sourcesXCoord, vector<int> sourcesId)
24  {
25  if(sourcesXCoord.at(0) <= 0 || sourcesXCoord.back() >= regionSize.at(0))
26  throw invalid_argument("X coords of sources should be in interval (0, width)");
27 
28  auto width = regionSize.at(0),
29  height = regionSize.at(1);
30 
31  //corner Points of rectangular region
32  vector<tet::Point> regionPoints{
33  tet::Point(width, 0),
34  tet::Point(width, height),
35  tet::Point(0, height),
36  tet::Point(0, 0)
37  };
38 
39  //Border Lines
40  vector<tet::MeshElement * > borderLines
41  {
42  new tet::Line(0, 1, boundariesId.at(0)),//indexes starts from zero
43  new tet::Line(1, 2, boundariesId.at(1)),
44  new tet::Line(2, 3, boundariesId.at(2))
45  };
46 
47  //Iterating over sources points
48  vector<tet::MeshElement *> sources;
49  unsigned i = 0;
50  for(; i < sourcesId.size(); ++i)
51  {
52  //Pushing left point
53  regionPoints.push_back(
54  tet::Point(sourcesXCoord[i] - eps/2, 0/*y coord*/));
55  sources.push_back(
56  new tet::PhysPoint(regionPoints.size() - 1 , sourcesId[i]));
57 
58  if(i < sourcesId.size())
59  borderLines.push_back(
60  new tet::Line(regionPoints.size() - 2, regionPoints.size() - 1,
61  boundariesId.at(3)));
62 
63 
64  //Pushing right point
65  regionPoints.push_back(
66  tet::Point(sourcesXCoord[i] + eps/2, 0/*y coord*/));
67  sources.push_back(
68  new tet::PhysPoint(regionPoints.size() - 1 , sourcesId[i]));
69 
70  //last one point is ommited
71  }
72  borderLines.push_back(
73  new tet::Line(regionPoints.size() - 1, 0, boundariesId[3]));
74 
75 
76  //passing values to meshBorder object
77  borderMesh.set_vertexes(regionPoints);
78  borderMesh.set_points(sources);
79  borderMesh.set_lines(borderLines);
80 
81  return *this;
82  }

190  double Border::GetSourceNormalAngle(int source_id) const
191  {
192  auto [p_left, p_right] = GetSourceVerticesIndexById(source_id);

200  v_left.get_coord(0) - adjacent_left.get_coord(0),
201  v_left.get_coord(1) - adjacent_left.get_coord(1));
202 
203  auto angle_right = River::Point::angle(
204  adjacent_right.get_coord(0) - v_right.get_coord(0),
205  adjacent_right.get_coord(1) - v_right.get_coord(1));
206 
207  return (angle_left + angle_right) / 2 + M_PI/2/*we are looking for perpendicular*/;
