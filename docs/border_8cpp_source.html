23  Border& Border::MakeRectangular(
24  vector<double> regionSize, vector<int> boundariesId,
25  vector<double> sourcesXCoord, vector<int> sourcesId)
26  {
27  //clear all data
28  vertices.clear();
29  sources.clear();
30  lines.clear();
31 
32  //checks
33  if(sourcesXCoord.at(0) <= 0 || sourcesXCoord.back() >= regionSize.at(0))
34  throw invalid_argument("X coords of sources should be in interval (0, width)");
35 
36  if(!is_sorted(sourcesXCoord.begin(), sourcesXCoord.end()))
37  throw invalid_argument("Sources X coords should be sorted! From smaller to greater");
38 
39  auto width = regionSize.at(0),
40  height = regionSize.at(1);
41 
42  //corner Points of rectangular region
43  vertices = {
44  Point{width, 0},
45  Point{width, height},
46  Point{0, height},
47  Point{0, 0}
48  };
49 
50  //Border Lines
51  lines = {
52  {0, 1, boundariesId.at(0)},
53  {1, 2, boundariesId.at(1)},
54  {2, 3, boundariesId.at(2)}
55  };
56 
57  //Iterating over sources points
58  for(long unsigned i = 0; i < sourcesId.size(); ++i)
59  {
60  vertices.push_back(Point{sourcesXCoord.at(i), 0});
61  auto index = vertices.size() - 1;
62  sources[sourcesId.at(i)] = index;
63  lines.push_back({index - 1, index, boundariesId.at(3)});
64  }
65  lines.push_back({vertices.size() - 1, 0, boundariesId.at(3)});
66 
67  return *this;
68  }
