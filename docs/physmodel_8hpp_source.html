126  inline double operator()(double x, double y) const
127  {
129  double result_area = 10000000/*some large area value*/;
130  for(auto& tip: tip_points)
131  {
132  auto r = (Point{x, y} - tip).norm();
133  auto exp_val = exp( -pow(r/refinment_radius, exponant)/2./sigma/sigma);
134  auto cur_area = min_area + (max_area - min_area)*(1. - exp_val)/(1. + exp_val);
135  if(result_area > cur_area)
136  result_area = cur_area;
137  }

333  Polar next_point(vector<double> series_params, double branch_lenght) const
334  {
335  //handle situation near biffurcation point, to reduce "killing/shading" one branch by another
336  auto eta_local = eta;
337  if(branch_lenght < growth_min_distance)
338  eta_local = 0;//constant growth of both branches.
339 
340  auto beta = series_params.at(0)/series_params.at(1),
341  dl = ds * pow(series_params.at(0), eta_local);
342  if(growth_type == 0)
343  {
344  double phi = -atan(2 / beta * sqrt(dl));
345  return {dl, phi};
346  }
347  else if(growth_type == 1)
348  {
349  auto dy = beta*beta/9*( pow(27/2*dl/beta/beta + 1, 2./3.) - 1),
350  dx = 2*sqrt( pow(dy, 3)/pow(beta,2) + pow(dy, 4) / pow(beta, 3));
351 
352  return Point{dx, dy}.getPolar();
353  }
354  else
355  throw invalid_argument("Invalid value of growth_type!");
356  }
