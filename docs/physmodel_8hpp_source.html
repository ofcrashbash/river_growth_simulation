156  auto r = (Point{x, y} - tip).norm();
157  auto exp_val = exp( -pow(r/refinment_radius, exponant)/2./sigma/sigma);

366  Polar next_point(vector<double> series_params, double branch_lenght, double max_a) const
367  {
368  //handle situation near bifurcation point, to reduce "killing/shading" one branch by another
369  auto eta_local = eta;
370  if(branch_lenght < growth_min_distance)
371  eta_local = 0;//constant growth of both branches.
372 
373  auto beta = series_params.at(0)/series_params.at(1),
374  dl = ds * pow(series_params.at(0)/max_a, eta_local);
375  if(growth_type == 0)
376  {
377  double phi = -atan(2 / beta * sqrt(dl));
378  return {dl, phi};
379  }
380  else if(growth_type == 1)
381  {
382  auto dy = beta*beta/9*( pow(27/2*dl/beta/beta + 1, 2./3.) - 1),
383  dx = 2*sqrt( pow(dy, 3)/pow(beta, 2) + pow(dy, 4) / pow(beta, 3));
384 
385  return ToPolar(Point{dx, dy}.rotate(-M_PI/2));
386  }
387  else
388  throw invalid_argument("Invalid value of growth_type!");
389  }
