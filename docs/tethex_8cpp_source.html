422 IncidenceMatrix::IncidenceMatrix(int n_vertices,
423  const std::vector<MeshElement*> &cells)
424  : dim(n_vertices)
425  , n_non_zero(0)
426  , row(nullptr)
427  , col(nullptr)
428 {
429  std::vector<int> *vec = new std::vector<int>[dim]; // for lower triangle
430  // look through all mesh cells
431  for (size_t cell = 0; cell < cells.size(); ++cell)
432  {
433  // look at all pairs of cell vertices
434  for (int i = 0; i < cells[cell]->get_n_vertices(); ++i)
435  {
436  const int ii = cells[cell]->get_vertex(i);
437  for (int j = 0; j < cells[cell]->get_n_vertices(); ++j)
438  {
439  const int jj = cells[cell]->get_vertex(j);
440  if (ii > jj) // we consider only lower triangle of matrix
441  {
442  // add to vector, if the vector doesn't contain this number
443  if (std::find(vec[ii].begin(), vec[ii].end(), jj) == vec[ii].end())
444  vec[ii].push_back(jj);
445  }
446  }
447  }
448  }
449 
450  // sorting the vectors
451  for (int i = 0; i < dim; ++i)
452  std::sort(vec[i].begin(), vec[i].end());
453 
454  // the number of non zero elements in each row of lower triangle
455  row = new int[dim + 1];
456  row[0] = 0;
457  for (int i = 0; i < dim; ++i)
458  row[i + 1] = row[i] + vec[i].size();
459 
460  n_non_zero = row[dim]; // the number of all non zero elements in lower triangle
461 
462  // numbers of non zero elements in lower triangle
463  col = new int[n_non_zero];
464  int k = 0;
465  for (int i = 0; i < dim; ++i)
466  {
467  for (size_t j = 0; j < vec[i].size(); ++j)
468  {
469  col[k] = vec[i][j];
470  k++;
471  }
472  }
473 
474  // free the memory
475  for (int i = 0; i < dim; ++i)
476  vec[i].clear();
477  delete[] vec;
478 }

491  int col_number) const
492 {
493  // because we seek in lower triangle, row must be bigger than col
494  expect(row_number > col_number,

496  d2s(row_number) + ", col_number = " + d2s(col_number) + "!");
497 
498  for (int i = row[row_number]; i < row[row_number + 1]; ++i)
499  {
500  if (col[i] == col_number)
501  return i;
502  }
503 
504  for (int i = row[col_number]; i < row[col_number + 1]; ++i)
505  {
506  if (col[i] == row_number)
507  return i;
508  }

783 void Mesh::read(const std::string &file)
784 {

787 
788  clean(); // free the memory for mesh elements
789 
790  std::string str;
791  in >> str; // the first string of Gmsh file is "$MeshFormat"
792  expect(str == "$MeshFormat", "The first string of the Gmsh file " + file +
793  " doesn't equal to \"$MeshFormat\". The actual string is \"" + str + "\"");
794 
795  // read the information about the mesh
796  double version;
797  int binary, dsize;
798  in >> version >> binary >> dsize;
799  // The function has been testing for meshes corresponding
800  // to Gmsh versions since 2.6.0, therefore
801  // in debug mode you'll have an exception if the mesh version is less than 2.2.
802  // There is no exception in release mode though.
803  // So to read the mesh with version 2.1 and less, check that DEBUG variable is set to 0.
804  // But NOTE that msh files of 1.0 format have absolutely different structure!
805  expect(version >= 2.2, "The version of Gmsh's mesh is too old (" + d2s(version) +
806  "). The library was tested for versions 2.2+.");
807  expect(dsize == sizeof(double), "The size of Gmsh's double (" + d2s(dsize) +
808  ") doesn't equal to size of double type (" + d2s(sizeof(double)) + ")");
809 
810  getline(in, str); // read some empty string
811 
812 // there is additional 1 (the number - one) in binary format
813  if (binary)
814  {
815  int one;
816  in.read(reinterpret_cast<char*>(&one), sizeof(int));
817  require(one == 1, "The binary one (" + d2s(one) + ") doesn't equal to 1!");
818  }
819 
820  // we make a map between serial number of the vertex and its number in the file.
821  // it will help us when we create mesh elements
822  std::map<int, int> vertices_map;
823 
824  // read lines of mesh file.
825  // if we face specific keyword, we'll treat the section.
826  while (in >> str)
827  {
828  if (str == "$PhysicalNames") // read the section of names of physical entities
829  {
830  int n_names;
831  in >> n_names;
832  getline(in, str);
833  physical_names.resize(n_names);
834  for (int i = 0; i < n_names; ++i)
835  getline(in, physical_names[i]);
836  }
837 
838  else if (str == "$Nodes") // read the mesh vertices
839  {
840  int n_vertices; // the number of all mesh vertices (that are saved in the file)
841  in >> n_vertices; // read that number
842  vertices.resize(n_vertices); // allocate the memory for mesh vertices
843  getline(in, str); // read some empty string
844 
845  int number; // the number of the vertex
846  // Cartesian coordinates of the vertex (Gmsh produces 3D mesh regardless
847  // its real dimension)
848  double coord[Point::n_coord];
849 
850  // read vertices
851  for (int ver = 0; ver < n_vertices; ++ver)
852  {
853  if (binary) // binary format
854  {
855  in.read(reinterpret_cast<char*>(&number), sizeof(int));
856  in.read(reinterpret_cast<char*>(coord), Point::n_coord * sizeof(double));
857  }
858  else // ASCII format
859  {
860  in >> number;
861  for (int i = 0; i < Point::n_coord; ++i)
862  in >> coord[i];
863  }
864  vertices[ver] = Point(coord); // save the vertex
865  vertices_map[number] = ver; // add the number of vertex to the map
866  }
867 
868  expect(n_vertices == vertices_map.size(),
869  "Vertices numbers are not unique: n_vertices = " + d2s(n_vertices) +
870  " vertices_map.size() = " + d2s(vertices_map.size()));
871 
872  } // read the vertices
873 
874  else if (str == "$Elements") // read the mesh elements
875  {
876  int n_elements; // the number of mesh elements
877  in >> n_elements; // read that number
878  getline(in, str); // empty string
879 
880  int number; // the number of the element [1, nElements]
881  int el_type; // the type of the element (1 - line, 2 - triangle, etc)
882  int n_tags; // the number of tags describing the element
883  int phys_domain; // the physical domain where the element takes place
884 // int elem_domain; // the elementary domain where the element takes place
885 // int partition; // the partition in which the element takes place
886 
887  // the map between the type of the element,
888  // and the number of nodes that describe it
889  std::map<int, int> type_nodes;
890  type_nodes[1] = 2; // 2-nodes line
891  type_nodes[2] = 3; // 3-nodes triangle
892  type_nodes[3] = 4; // 4-nodes quadrangle
893  type_nodes[4] = 4; // 4-nodes tetrahedron
894  type_nodes[5] = 8; // 8-nodes hexahedron
895  type_nodes[15]= 1; // 1-node point
896 
897  {
898  for (int el = 0; el < n_elements; ++el)
899  {
900  in >> number >> el_type >> n_tags;
901  std::vector<int> data(n_tags); // allocate the memory for some data
902  for (int i = 0; i < n_tags; ++i) // read this information
903  in >> data[i];
904  phys_domain = (n_tags > 0) ? data[0] : 0; // physical domain - the most important value
905 // elem_domain = (n_tags > 1) ? data[1] : 0; // elementary domain
906 // partition = (n_tags > 2) ? data[2] : 0; // partition (Metis, Chaco, etc)
907  data.clear(); // other data isn't interesting for us
908 
909  // how many vertices (nodes) describe the element
910  std::map<int, int>::const_iterator el_type_iter =
911  type_nodes.find(el_type);
912 
913  require(el_type_iter != type_nodes.end(),
914  "This type of the Gmsh's element (" + d2s(el_type) +
915  ") in the mesh file \"" + file + "\" is unknown!");
916 
917  const int n_elem_nodes = el_type_iter->second; // the number of nodes
918  std::vector<int> nodes(n_elem_nodes); // allocate memory for nodes
919  for (int i = 0; i < n_elem_nodes; ++i)
920  {
921  in >> nodes[i]; // read the numbers of nodes
922  // vertices can be numerated not sequentially (or not from 0)
923  nodes[i] = vertices_map.find(nodes[i])->second;
924  }
925 
926  switch (el_type)
927  {
928  case 15: // 1-node point
929  points.push_back(new PhysPoint(nodes, phys_domain));
930  break;
931  case 1: // 2-nodes line
932  lines.push_back(new Line(nodes, phys_domain));
933  break;
934  case 2: // 3-nodes triangle
935  triangles.push_back(new Triangle(nodes, phys_domain));
936  break;
937  case 3: // 4-nodes quadrangle
938  quadrangles.push_back(new Quadrangle(nodes, phys_domain));
939  break;
940  default:
941  require(false, "Unknown type of the Gmsh's element (" +
942  d2s(el_type) + ") in the file " + file + "!");
943  }
944 
945  nodes.clear();
946  }
947 
948  // check some expectations
949  expect(number == n_elements, "The number of the last read Gmsh's "
950  "element (" + d2s(number) + ") is not equal to the amount of "
951  "all elements in the mesh (" + d2s(n_elements) + ")!");
952 
953  } // ASCII format
954 
955  // requirements after reading elements
956  require(!triangles.empty() || !quadrangles.empty(),
957  "There are no any 2D or 3D elements in the mesh!");
958 
959  // we prevent mixing of simplices and bricks in one mesh.
960  // at least for the moment
961  if (!triangles.empty())
962  require(quadrangles.empty(), "There are simplices "
963  "and bricks in the same mesh. It's prohibited. Mesh file " + file);
964 
965  } // read the elements
966  }
967 
968  in.close(); // close the file
969 }

1058  int n_old_vertices,

1061 {
1062  // quadrangles generation
1063  std::vector<int> quadrangle_vertices(Quadrangle::n_vertices);
1064 
1065  // we need to numerate edges of boundary triangles
1066  // numerate_edges = true - the case of 3D mesh, when we numerate edges of boundary triangles,
1067  // but in this case we mustn't numerate edges themselves,
1068  // because they were already numerated during process of numeration edges of tetrahedra,
1069  // numerate_edges = false - the case of 2D mesh, when we have already numerated edges of triangles,
1070  // so we don't need to do that again
1071  if (numerate_edges)
1072  // third parameter - whether we need to initialize the vector of all edges of the mesh,
1073  // no we shouldn't
1074  edge_numeration(triangles, incidence_matrix, false);
1075 
1076  for (size_t tri = 0; tri < triangles.size(); ++tri)
1077  {
1078  for (int ver = 0; ver < Triangle::n_vertices; ++ver)
1079  {
1080  // current vertex
1081  const int cur_vertex = triangles[tri]->get_vertex(ver);
1082 
1083  // we are looking for 2 edges that contain current vertex
1084  std::vector<int> seek_edges;
1085  for (int edge = 0; edge < Triangle::n_edges; ++edge)
1086  {
1087  const int cur_edge = triangles[tri]->get_edge(edge);
1088  if (edges[cur_edge]->contains(cur_vertex))
1089  seek_edges.push_back(cur_edge);
1090 
1091  }
1092  expect(seek_edges.size() == 2,
1093  "The number of edges to which every vertex belongs must be equal to 2");
1094

1097  quadrangle_vertices[1] = n_old_vertices + seek_edges[0];

1100  {
1101  // 3D case - boundary triangles
1102  quadrangle_vertices[2] = n_old_vertices + edges.size() +
1103  find_face_from_two_edges(seek_edges[0],
1104  seek_edges[1],
1105  incidence_matrix,
1106  edge_vertex_incidence);
1107  }
1108  else
1109  {
1110  // 2D case - triangles themselves
1111  quadrangle_vertices[2] = n_old_vertices + edges.size() + tri;
1112  }
1113  quadrangle_vertices[3] = n_old_vertices + seek_edges[1];
1114 
1115  seek_edges.clear();

1118  // and it's important not to mix these 2 directions.
1119  // so, we need additional check as deal.II authors do.
1120  change_vertices_order(2, vertices, quadrangle_vertices);
